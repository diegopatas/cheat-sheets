```bash
# COMMAND MODE
:<type cmd>
:w          # save (write)
:savas      # save {file} as
:clo        # close current pane
:q			# just quit
:q!			# quit, don't apply changes
:wq			# save and quit
:h {y}      # open help for keyword {y}
:ter        # open terminal (type ctrl + d to close)
K           # open man for word under cursor
```
