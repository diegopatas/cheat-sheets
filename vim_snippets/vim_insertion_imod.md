```sh
i       # insert (starts insert mode)
I		# insert first (starts insert mode)
a		# append
A		# append last
o		# <CR> insert next line
O		# <CR> insert sameline line
ea      # insert at the end of the word
esc     # exit insert mode
<C-o>   # (in insert mode) switch to one strike normal mode ex.:<C-o>dd
```
