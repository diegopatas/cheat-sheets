```bash
# VISUAL MODE (v or V)
o                # move to other end of marked area
ctrl + v         # start visual block
O                # move to other corner of block
aw               # mark a word
ab               # mark a block w/ () include ()
a(
aB               # mark a block w/ {} include {}
a{
at               # mark a block w/ <>
ib               # mark a inner block w/ () exclude ()
i(
iB               # mark a inner block w/ {} exclude {}
i{
it               # mark a inner block w/ <>
ESC              # exit mode
```
