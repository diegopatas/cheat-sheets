```bash
yy         # yank line
2yy        # yank 2 lines
yw         # yank word
yiw        # yank inner word
yaw        # yank word + space after
y$ (or Y)  # yank to EOL
p          # paste in line bellow
P          # paste in line above
gp         # paste bellow + set cursor bellow
gP         # paste above + set cursor bellow
dd         # delete (cut) line
2dd        # delete (cut) 2 lines
dw         # delete (cut) word
diw        # delete inner word
daw        # delete word + space after
d$ (or D)  # delete to the end of the line
x          # delete (cut) char
```
